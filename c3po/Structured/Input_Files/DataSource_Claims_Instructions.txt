WITH base_data AS (

  FROM claims_data
  WHERE indication = '[INDICATION]'

2. Denominator CTE (All Regimens):
, denominator_monthly AS (

    SUM(Monthly_Adjusted_Count) AS total_count

  ORDER BY Year_Month

3. Numerator CTE (Specific Regimen):
, numerator_monthly AS (

    Year_Month,

  GROUP BY Year_Month
  ORDER BY Year_Month
)

- Note - Always execute the denominator filters first and then the numerator filters

ROLLING 3 MONTH CALCULATIONS - 

, denominator_r3m AS (
  SELECT
    d1.Year_Month,
    SUM(d2.total_count) AS r3m_total_count
  FROM denominator_monthly d1
  JOIN denominator_monthly d2 
    ON d2.Year_Month <= d1.Year_Month 
    AND d2.Year_Month >= DATE_FORMAT(ADD_MONTHS(TO_DATE(CAST(d1.Year_Month AS STRING), 'yyyyMM'), -2), 'yyyyMM')
  GROUP BY d1.Year_Month
  HAVING COUNT(*) = 3  -- Ensure we have exactly 3 months
  ORDER BY d1.Year_Month
)

, numerator_r3m AS (
  SELECT
    n1.Year_Month,
    SUM(n2.regimen_count) AS r3m_regimen_count
  FROM numerator_monthly n1
  JOIN numerator_monthly n2 
    ON n2.Year_Month <= n1.Year_Month 
    AND n2.Year_Month >= DATE_FORMAT(ADD_MONTHS(TO_DATE(CAST(n1.Year_Month AS STRING), 'yyyyMM'), -2), 'yyyyMM')
  GROUP BY n1.Year_Month
  HAVING COUNT(*) = 3  -- Ensure we have exactly 3 months
  ORDER BY n1.Year_Month
)

Final R3M Share Calculation - 

SELECT
  r.Year_Month,
  r.r3m_regimen_count,
  d.r3m_total_count,
  (r.r3m_regimen_count / d.r3m_total_count) * 100 AS r3m_share_percentage
FROM numerator_r3m r
JOIN denominator_r3m d ON r.Year_Month = d.Year_Month
WHERE SUBSTRING(r.Year_Month, 1, 4) = '[REQUESTED_YEAR]'  -- Filter for requested year
ORDER BY r.Year_Month
