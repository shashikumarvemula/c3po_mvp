    def create_unique_truncated_labels(full_labels, base_chars=20):
        """Create truncated labels that extend length to remain unique"""
        truncated_labels = []
        
        for i, label in enumerate(full_labels):
            if not label:
                truncated_labels.append("Unknown")
                continue
                
            label_str = str(label).strip()
            
            # Start with base truncation
            current_length = base_chars
            truncated = label_str[:current_length]
            
            # Check if this truncation conflicts with any previous labels
            while current_length <= len(label_str):
                # Check if this truncated version matches any already created labels
                conflicts = False
                for existing_label in truncated_labels:
                    if existing_label.replace("...", "") == truncated and existing_label != "Unknown":
                        conflicts = True
                        break
                
                # If no conflict, use this length
                if not conflicts:
                    break
                    
                # Extend by 5 characters to find unique part
                current_length += 5
                truncated = label_str[:current_length]
            
            # Add ellipsis if truncated
            if current_length < len(label_str):
                final_label = truncated + "..."
            else:
                final_label = label_str  # Use full text if short enough
                
            truncated_labels.append(final_label)
        
        return truncated_labels
